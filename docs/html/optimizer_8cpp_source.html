12 std::pair<size_t, float> determine_lod_count_and_ratio(const PositionArray &positions) {
13 constexpr int maxlods = 3;
14
15 float lod_scale = meshopt_simplifyScale((float*)positions.data(), positions.size(), sizeof(Position));
16 size_t lod_count = 0;
17
18 if (lod_scale > 1) {
19 MR_WARNING("LOD Scale greater than 1. Needs investigation");
20 lod_scale = 0.1;
21 }
22
23 // we want any mesh to have at least 47 triangles
24 while (positions.size() * std::pow(lod_scale, lod_count) / 3 >= 47) {
25 lod_count++;
26 }
27
28 if (lod_count > maxlods) {
29 lod_scale = std::pow(lod_scale, lod_count / (float)maxlods);
30 lod_count = maxlods;
31 }
32
33 return {lod_count, lod_scale};
34 }

50 const size_t target_index_count = original_index_count * lod_scale / 3 * 3;
51 const bool is_sparse = lod_scale <= 4 / std::sqrt(original_index_count);
52
53 float lod_error = 0.f;
54
55 result_indices.resize(original_index_count);
56
57 uint32_t options

66 (float*)positions.data(), positions.size(), sizeof(Position),
67 target_index_count, target_error,
68 options, &lod_error)
69 );
70
71 MR_INFO("LOD {}:", lod_index);
72 MR_INFO("\tTarget index count: {}", target_index_count);
73 MR_INFO("\tIs sparse: {}", is_sparse);
74 MR_INFO("\tError: {}", lod_error);
75
76 meshopt_optimizeVertexCache(result_indices.data(), result_indices.data(), result_indices.size(), positions.size());
77
78 result_shadow_indices.resize(result_indices.size());

91 Mesh optimize(Mesh mesh) {
92 if (mesh.attributes.empty()) {
93 MR_WARNING(
94 "Mesh doesn't contain attributes but they are being taken into account by `optimize` function."
95 "Consider adding attribute-less path in optimize"
96 );
97 mesh.attributes.resize(mesh.positions.size());
98 }
99
100 Mesh result;
101 result.transforms = mesh.transforms;
102 result.name = mesh.name;
103
104 std::array streams = {
105 meshopt_Stream{mesh.positions.data(), sizeof(Position), sizeof(Position)},
106 meshopt_Stream{mesh.attributes.data(), sizeof(VertexAttributes), sizeof(VertexAttributes)},
107 };
108
109 auto [count, ratio] = determine_lod_count_and_ratio(mesh.positions);
110 result.lods.resize(count+1);
111
112 // improve vertex locality
113 meshopt_optimizeVertexCache(mesh.lods[0].indices.data(), mesh.lods[0].indices.data(), mesh.lods[0].indices.size(), mesh.positions.size());
114
115 // optimize overdraw
116 meshopt_optimizeOverdraw(mesh.lods[0].indices.data(), mesh.lods[0].indices.data(), mesh.lods[0].indices.size(),
117 (float*)mesh.positions.data(), mesh.positions.size(), sizeof(Position), 1.05f);
118
119 IndexArray remap;
120 remap.resize(mesh.lods[0].indices.size());
121 size_t vertex_count = meshopt_generateVertexRemapMulti(
122 remap.data(),
123 mesh.lods[0].indices.data(),
124 mesh.lods[0].indices.size(),
125 mesh.positions.size(),
126 streams.data(),
127 streams.size());
128
129 result.lods[0].indices.resize(mesh.lods[0].indices.size());
130 result.positions.resize(vertex_count);
131 result.attributes.resize(vertex_count);
132
133 meshopt_remapIndexBuffer(result.lods[0].indices.data(), mesh.lods[0].indices.data(), mesh.lods[0].indices.size(), remap.data());
134 meshopt_remapVertexBuffer(result.positions.data(), mesh.positions.data(), mesh.positions.size(), sizeof(Position), remap.data());
135 meshopt_remapVertexBuffer(result.attributes.data(), mesh.attributes.data(), mesh.positions.size(), sizeof(VertexAttributes), remap.data());
136

139 result.lods[0].indices.data(), result.lods[0].indices.size(),
140 result.positions.size());
141
142 // NOTE: we run remap functions the second time as recommended by docs
143 meshopt_remapIndexBuffer(result.lods[0].indices.data(), result.lods[0].indices.data(), result.lods[0].indices.size(), remap.data());
144 meshopt_remapVertexBuffer(result.positions.data(), result.positions.data(), result.positions.size(), sizeof(Position), remap.data());
145 meshopt_remapVertexBuffer(result.attributes.data(), result.attributes.data(), result.attributes.size(), sizeof(VertexAttributes), remap.data());
146
147 result.lods[0].shadow_indices.resize(result.lods[0].indices.size());
148 meshopt_generateShadowIndexBufferMulti(result.lods[0].shadow_indices.data(),
149 result.lods[0].indices.data(), result.lods[0].indices.size(),
150 result.positions.size(),
151 streams.data(), streams.size());
152
153 // LOD generation
154 for (int i = 1; i < count+1; i++) {
